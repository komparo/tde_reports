datasets <- inputs[["datasets"]] %>% 
  as_tibble() %>% 
  mutate(
    expression = map(expression, read.table, row.names = 1, header = TRUE, sep = ","),
    n_cells = map_int(expression, nrow),
    n_features = map_int(expression, ncol)
  )
datasets %>% 
  select(n_cells, n_features) %>% 
  gather("feature", "n") %>% 
  ggplot(aes(feature, n)) +
   geom_boxplot() +
  theme_bw()
